                    package com.Ejercicio7;


                    import java.io.IOException;
                    import java.util.PrimitiveIterator;
                    import java.util.Scanner;
                    
                    public class Ejercicio7 {
                    
                        private Scanner input = new Scanner(System.in);
                       private int option;
                       private boolean flag = true;
                       private String name;
                    
                       private double weight, height, imc;
                    
                        public void Ejercicio7 () throws IOException {
                    
                            do {
                                menu();
                                option = captureDataInt();
                                input.nextLine(); //Limpiar el buffer de entrada después de leer el número entero
                                flag = evaluateOption(option);
                            }while (flag == true);
                        }
                    
                        private void menu(){
                            System.out.println("App de la Secretaría de Salud Municipal ");
                            System.out.println("1. Calcular el IMC (Índice de masa corporal)");
                            System.out.println("0. Salir de la App");
                        }
                    
                        private void askName(){
                            System.out.println("Por favor, digite el nombre del usuario");
                        }
                    
                        private void askWeight(){
                            System.out.println("Por favor, digite el peso del usuario en kilogramos");
                        }
                    
                        private void askHeight(){
                            System.out.println("Por favor, digite la estatura del usuario en metros");
                        }
                    
                        private String captureDataString(){
                            return input.nextLine();
                        }
                    
                        private int captureDataInt(){
                            return input.nextInt();
                        }
                    
                        private double captureDataDouble(){
                            return input.nextDouble();
                        }
                    
                        private boolean evaluateOption(int option) throws IOException {
                            switch (option){
                                case 1:
                                    captureData();
                                    imc();
                                    rank();
                                    System.out.println("Digita, enter para volver al menu principal");
                                    System.in.read();
                                    return true;
                                case 0:
                                    return false;
                                //break;
                                default:
                                    System.out.println("Opción incorrecta");
                                    System.out.println("Digita, enter para volver al menu principal");
                                    System.in.read();
                                    return true;
                            }
                        }
                    
                        private void captureData(){
                            askName();
                            name = captureDataString();
                            askWeight();
                            weight = captureDataDouble();
                            askHeight();
                            height = captureDataDouble();
                        }
                    /*  Formula de IMC
                        IMC=Peso(Kg)/Altura^2(m)
                        Peso inferior al normal	Menos de 18.5
                        Peso Normal	18.5 - 24.9
                        Peso superior al normal	25.0 - 29.9
                        Obesidad	Más de 30.0
                        */
                        private void imc(){
                            imc=(weight/(Math.pow(height,2)));
                            System.out.println("El IMC de "+name+" es de: "+imc);
                        }
                    
                        private void rank (){
                            if (imc<18.5){
                                System.out.println(name+" se encuentra bajo de peso ");
                            }
                            if((imc<=24.9)&&(imc>=18.5)){
                                System.out.println(name+" se encuentra normal de peso ");
                            }
                            if((imc<=29.9)&&(imc>=25)){
                                System.out.println(name+" se encuentra en sobrepeso ");
                            }
                            if(imc>=30){
                                System.out.println(name+" se encuentra obeso ");
                            }
                        }
                    }
