                    package com.Ejercicio5;

                    import java.io.IOException;
                    import java.util.Scanner;
                    
                    public class Ejercicio5 {
                        private Scanner input = new Scanner(System.in);
                        private int option;
                        private boolean flag = true;
                        private int [][] matrix;
                    
                        private int i,row,col,product;
                    
                        public void Ejercicio5 () throws IOException {
                            matrix = new int[10][10];
                            do {
                                menu();
                                option = captureDataInt();
                                input.nextLine(); //Limpiar el buffer de entrada después de leer el número entero
                                flag = evaluateOption(option);
                            }while (flag == true);
                        }
                    
                        private void menu(){
                            System.out.println("1. Tabla y resultados ");
                            System.out.println("0. Salir de la App");
                        }
                    
                        private int captureDataInt(){
                            return input.nextInt();
                        }
                    
                        private boolean evaluateOption(int option) throws IOException {
                            switch (option){
                                case 1:
                                    printTable();
                                    resultsMatrix();
                                    userData();
                                    System.out.println("Resultado de la celda ="+matrix[row][col]);
                                    System.out.println("Digita, enter para volver al menu principal");
                                    System.in.read();
                                    return true;
                                case 0:
                                    return false;
                                //break;
                                default:
                                    System.out.println("Opción incorrecta");
                                    System.out.println("Digita, enter para volver al menu principal");
                                    System.in.read();
                                    return true;
                            }
                        }
                    
                        private void userData(){
                            System.out.println("Insertar la fila y columna de la celda acorde a la tabla de la cual desea ver el resultado");
                            do {
                                System.out.println("Digite el número de la fila (de 0 a 9)");
                                row=captureDataInt();
                                if ((row>9)||(row<0)){
                                    System.out.println("Número fuera de rango, intenta de nuevo");
                                }
                            }while ((row>9)||(row<0));
                    
                            do {
                                System.out.println("Digite el número de la columna (de 0 a 9)");
                                col=captureDataInt();
                                if ((col>9)||(col<0)){
                                    System.out.println("Número fuera de rango, intenta de nuevo");
                                }
                            }while ((col>9)||(col<0));
                        }
                    
                        private void resultsMatrix(){
                            for (int col = 0; col < 10; col++) {
                                for (int row = 0; row <10; row++) {
                                    product=(row+1)*(col+1);
                                    matrix[row][col]=product;
                                }
                            }
                        }
