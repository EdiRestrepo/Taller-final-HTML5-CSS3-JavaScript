La Droguería Mi Salud presta sus servicios en la localidad de Suba y requiere una aplicación para poder facturar los productos que vende a sus clientes y para ello, los productos tienen unas características que deben indicársele al cliente para que pueda escoger el producto a comprar.                     package com.Ejercicio5;


                    import java.io.IOException;
                    import java.util.Scanner;
                    
                    public class Ejercicio5 {
                    
                        private Scanner input = new Scanner(System.in);
                        private String name;
                       private int option;
                       private boolean flag = true;
                       private String[] product;
                       private String[] productPrice;
                       private String[] returns;
                    
                        public void Ejercicio5 () throws IOException {
                            returns = new String[3];
                            product = new String[3];
                            productPrice = new String[3];
                            product[0]="Pañitos humedos para el cuidado de la piel X 100 Unidades";
                            product[1]="Protector solar x 1000ml";
                            product[2]="Acetaminofén de 500mg para el dolor de cabeza, tableta por 10 pastillas";
                            productPrice[0]="5000";
                            productPrice[1]="2000";
                            productPrice[2]="1500";
                    
                            System.out.println("APP Droguería Mi Salud");
                            askName();
                            name = captureDataString();
                            System.out.println("Hola "+name+", bienvenido a su Droguería Mi Salud Suba");
                            do {
                                menu();
                                option = captureDataInt();
                                flag = evaluateOption(option);
                            }while (flag == true);
                        }
                    
                        private void menu(){
                            System.out.println("Por favor, digite un número de acuerdo a lo que desea hacer");
                            System.out.println("1. Comprar producto");
                            System.out.println("2. Consultar precio por producto");
                            System.out.println("3. Devolución de producto");
                            System.out.println("0. Salir de la App");
                        }
                    
                        private void askName(){
                            System.out.println("Por favor, digite su nombre de usuario");
                        }
                    
                        private String captureDataString(){
                            return input.nextLine().replace("\\n", "");
                        }
                    
                        private int captureDataInt(){
                            return Integer.parseInt(input.nextLine());
                        }
                    
                        private boolean evaluateOption(int option) throws IOException {
                            switch (option){
                                case 1:
                                    System.out.println("Por favor, digite un número de acuerdo al producto que desea comprar");
                                    products();
                                    option = captureDataInt();
                                    buyProduct(option);
                                    return true;
                                case 2:
                                    productsPrice();
                                    System.out.println("Digita, enter para volver al menu principal");
                                    System.in.read();
                                    return true;
                                case 3:
                                    System.out.println("Por favor, digite un número de acuerdo al producto que va a devolver");
                                    products();
                                    option = captureDataInt();
                                    returnProduct(option);
                                    return true;
                                case 0:
                                    return false;
                                //break;
                                default:
                                    System.out.println("Opción incorrecta");
                                    System.out.println("Digita, enter para volver al menu principal");
                                    System.in.read();
                                    return true;
                            }
                        }
                    
                        private void products(){
                            System.out.println("1. "+product[0]);
                            System.out.println("2. "+product[1]);
                            System.out.println("3. "+product[2]);
                        }
                    
                        private void productsPrice(){
                            System.out.println("1. "+product[0]+"= $"+productPrice[0]);
                            System.out.println("2. "+product[1]+"= $"+productPrice[1]);
                            System.out.println("3. "+product[2]+"= $"+productPrice[2]);
                        }
                    
                        private void buyProduct(int option) throws IOException {
                            switch (option){
                                case 1:
                                    System.out.println("Valor a pagar: $"+productPrice[0]);
                                    System.out.println(name+", gracias por su compra, hasta pronto");
                                    System.out.println("Digita, enter para volver al menu principal");
                                    System.in.read();
                                    break;
                                case 2:
                                    System.out.println("Valor a pagar: $"+productPrice[1]);
                                    System.out.println(name+", gracias por su compra, hasta pronto");
                                    System.out.println("Digita, enter para volver al menu principal");
                                    System.in.read();
                                    break;
                                case 3:
                                    System.out.println("Valor a pagar: $"+productPrice[2]);
                                    System.out.println(name+", gracias por su compra, hasta pronto");
                                    System.out.println("Digita, enter para volver al menu principal");
                                    System.in.read();
                                    break;
                                default:
                                    System.out.println("Opción incorrecta");
                                    System.out.println("Digita, enter para volver al menu principal");
                                    System.in.read();
                            }
                        }
                        private void returnProduct(int option) throws IOException {
                            switch (option){
                                case 1:
                                    System.out.println("Por favor, digite la razon de la devolución del producto");
                                    returns[0]=captureDataString();
                                    System.out.println("Se ha registrado la razon de la devolucion del producto");
                                    System.out.println("Se devuelve al cliente "+name+" un valor de: $"+productPrice[0]);
                                    System.out.println("Digita, enter para volver al menu principal");
                                    System.in.read();
                                    break;
                                case 2:
                                    System.out.println("Por favor, digite la razon de la devolución del producto");
                                    returns[1]=captureDataString();
                                    System.out.println("Se ha registrado la razon de la devolucion del producto");
                                    System.out.println("Se devuelve al cliente "+name+" un valor de: $"+productPrice[1]);
                                    System.out.println("Digita, enter para volver al menu principal");
                                    System.in.read();
                                    break;
                                case 3:
                                    System.out.println("Por favor, digite la razon de la devolución del producto");
                                    returns[2]=captureDataString();
                                    System.out.println("Se ha registrado la razon de la devolucion del producto");
                                    System.out.println("Se devuelve al cliente "+name+" un valor de: $"+productPrice[2]);
                                    System.out.println("Digita, enter para volver al menu principal");
                                    System.in.read();
                                    break;
                                default:
                                    System.out.println("Opción incorrecta");
                                    System.out.println("Digita, enter para volver al menu principal");
                                    System.in.read();
                            }
                        }
                    }
