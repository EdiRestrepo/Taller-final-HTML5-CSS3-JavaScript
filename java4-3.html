                    package com.Ejercicio3;

                    import java.io.IOException;
                    import java.util.Random;
                    import java.util.Scanner;
                    
                    public class Ejercicio3 {
                        private Scanner input = new Scanner(System.in);
                        private int option;
                        private boolean flag = true;
                    
                        private int i,counter,j, quantityNumbers;
                    
                        public void Ejercicio3 () throws IOException {
                            quantityNumbers=0;
                            do {
                                menu();
                                option = captureDataInt();
                                input.nextLine(); //Limpiar el buffer de entrada después de leer el número entero
                                flag = evaluateOption(option);
                            }while (flag == true);
                        }
                    
                        private void menu(){
                            System.out.println("1. Imprimir los números primos del 1 al 1000");
                            System.out.println("0. Salir de la App");
                        }
                    
                        private int captureDataInt(){
                            return input.nextInt();
                        }
                    
                        private boolean evaluateOption(int option) throws IOException {
                            switch (option){
                                case 1:
                                    primeNumbers();
                                    System.out.println("Digita, enter para volver al menu principal");
                                    System.in.read();
                                    return true;
                                case 0:
                                    return false;
                                //break;
                                default:
                                    System.out.println("Opción incorrecta");
                                    System.out.println("Digita, enter para volver al menu principal");
                                    System.in.read();
                                    return true;
                            }
                        }
                    
                        private void primeNumbers(){
                            for (int i = 1; i <= 1000; i++) {
                                j=1;
                                counter=0;
                                while (j<=i){
                                    if(i%j==0){
                                        counter++;
                                    }
                                    j++;
                                }
                                if(counter==2){
                                    System.out.println("El numero "+i+" es primo");
                                    quantityNumbers++;
                                }
                            }
                            System.out.println("Cantidad de numeros primos: "+quantityNumbers);
                        }
                    }
