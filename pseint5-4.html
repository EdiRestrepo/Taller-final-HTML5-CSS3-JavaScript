Exercise 4

Enunciado:
Organizar todos los puntos del taller de condicionales en modo funciones.

4. La video tienda que presta sus servicios de alquiler de películas a los usuarios del barrio el Porvenir, requiere de una aplicación que permita registrar el alquiler de las películas que adquieren sus usuarios. Para cada usuario se debe permitir la opción de alquilar película, consultar películas disponibles y recibir película en la video tienda con la opción de realizar anotaciones sobre estas si se llegan a presentar daños u otra novedad sobre la película. 


                //Recibir Película
                    SubProceso recibirPelicula (a)
                        Definir opcion Como Entero;
                        Limpiar Pantalla;
                        Escribir "¿Alguna novedad al recibir la película (daño u otra novedad)?";
                        Escribir "";
                        Escribir "1. Si";
                        Escribir "2. No";
                        Escribir "";
                        Escribir "Digita 1 si hay alguna novedad, de lo contrario digita 2 para salir del sistema";
                        Leer opcion;
                        Limpiar Pantalla;
                        Si opcion = 1 Entonces
                            Escribir "Digite cual es novedad";
                            Leer a;
                            Escribir "Se ha registrado la novedad: ", a;
                            Escribir "Hasta pronto";
                        SiNo
                            Escribir "Se ha guardado la información sin ninguna novedad";
                            Escribir "Hasta pronto";
                            
                        FinSi
                    FinSubProceso
                    
                    //Consultar películas disponibles
                    SubProceso peliculasDisponibles ()
                        Definir opcion Como Entero;
                        Limpiar Pantalla;
                        Escribir "Lista de Películas disponibles";
                        Escribir "";
                        Escribir "1. Película 1";
                        Escribir "2. Película 2";
                        Escribir "3. Película 3";
                        Escribir "4. Película 4";
                        Escribir "5. Película 5";
                        Escribir "";
                        Escribir "Regresa al menú principal si deseas alquilar alguna de nuestras películas";
                    FinSubProceso
                    
                    
                    //Alquilar Película
                    SubProceso alquilarPelicula (a Por Valor)
                        Definir opcion Como Entero;
                        Limpiar Pantalla;
                        Escribir "Lista de Películas disponibles";
                        Escribir "";
                        Escribir "1. Película 1";
                        Escribir "2. Película 2";
                        Escribir "3. Película 3";
                        Escribir "4. Película 4";
                        Escribir "5. Película 5";
                        Escribir "";
                        Escribir "Ingrese un número, de a acuerdo al nombre de la película que deseas alquilar";
                        Leer opcion;
                        Limpiar Pantalla;
                        
                        Segun opcion Hacer
                            1: 
                                Escribir "Hola ",a," ", "acabas de adquirir la Película 1, esperamos la disfrutes y regreses pronto"; 
                            2: 
                                Escribir "Hola ",a," ", "acabas de adquirir la Película 2, esperamos la disfrutes y regreses pronto";
                            3: 
                                Escribir "Hola ",a," ", "acabas de adquirir la Película 3, esperamos la disfrutes y regreses pronto"; 
                            4: 
                                Escribir "Hola ",a," ", "acabas de adquirir la Película 4, esperamos la disfrutes y regreses pronto"; 
                            5: 
                                Escribir "Hola ",a," ", "acabas de adquirir la Película 5, esperamos la disfrutes y regreses pronto"; 
                            De Otro Modo:
                                Escribir "La opción elegida no existe";
                                Escribir "Intenta de nuevo";
                                    FinSegun
                    FinSubProceso
                    
                    
                    
                    //Menú principal
                    SubProceso respuesta<-menu (a,b)
                        Definir respuesta Como Entero;
                        Limpiar Pantalla;
                        Si ConvertirANumero(a)>=14 Entonces
                            Escribir "Hola ", b,", Bienvenido a su video tienda Porvenir";
                            Escribir "";
                            Escribir "Ingrese un número del 1 al 4 de acuerdo a lo que desea hacer";
                            Escribir "";
                            Escribir "1. Alquilar película";
                            Escribir "2. Consultar películas disponbles";
                            Escribir "3. Recibir película en la video tienda";
                            Escribir "4. Salir de la App";
                        SiNo
                            Escribir "Hola ",b," ", "lo sentimos, pero no esta disponible el alquiler de peliculas a menores de 14 años de edad";
                            respuesta<-1;
                        FinSi
                    FinSubProceso
                    
                    //Ingreso datos usuario
                    SubProceso datos <- capturaDato (mensaje Por Valor)
                        Definir datos Como Caracter;
                        Escribir mensaje;
                        Leer datos;
                    FinSubProceso
                    
                    //App tienda alquiler películas
                    Proceso Taller5_4_Funcion_Condicional
                        
                        Definir usuario, edad, anotaciones Como Caracter;
                        Definir opcion, respuesta Como Entero;
                        
                        anotaciones<-"";
                        Escribir "APP video tienda alquiler de películas Porvenir";
                        Escribir "";
                        Escribir "Esta prohibido el alquiler a menores de 14 años de edad";
                        Escribir  "";
                        usuario<-capturaDato("Ingrese su nombre de usuario");
                        edad<-capturaDato("Ingrese su edad");
                        
                        respuesta<-menu(edad, usuario);
                        
                        Si respuesta=1 Entonces
                            Escribir"";
                        SiNo
                            Leer opcion;
                            
                            Segun opcion Hacer
                                1:
                                    alquilarPelicula(usuario);
                                2:
                                    peliculasDisponibles();
                                3:
                                    recibirPelicula(anotaciones);
                                4:
                                    Escribir "El usuario ",usuario," Ha salido de la App";
                                De Otro Modo:
                                    Escribir "La opción elegida no existe";
                            FinSegun
                        
                        FinSi
                        
                    FinProceso
