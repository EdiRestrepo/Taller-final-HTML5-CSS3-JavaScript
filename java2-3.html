Exercise 3

Enunciado:
Realizar un programa en el cual se solicite el nombre, apellidos y edad de la persona. Si la persona es mayor o igual a 18 años, entonce se deberá imprimir en pantalla [Nombre completo] usted es mayor de edad, por lo tanto puede entrar a la fiesta. Si la edad de la persona es menor que 18 años, entonces, deberá imprimirse el siguiente mensaje: [Nombre completo] usted es menor de edad, por lo tanto, no puede entrar a la fiesta, por favor devuélvase a su casa.

                    package com.Ejercicio3;

                    import java.util.Scanner;
                    
                    public class Ejercicio3 {
                    
                        private Scanner input;
                        private int age;
                        private String name;
                        private String surname;
                    
                        public void Ejercicio3() {
                            input = new Scanner(System.in);
                            askName();
                            name = captureDataString();
                    
                            askSurname();
                            surname = captureDataString();
                    
                            askAge();
                            age = captureDataInt();
                            evaluateAge();
                        }
                    
                        private void askName(){
                            System.out.println("Por favor, digite su nombre");
                        }
                        private void askSurname(){
                            System.out.println("Por favor, digite su apellido");
                        }
                        private void askAge () {
                            System.out.println("Por favor, digite su edad");
                        }
                    
                        private int captureDataInt () {
                            return input.nextInt();
                        }
                        private String captureDataString(){
                            return input.nextLine();
                        }
                    
                        private void evaluateAge(){
                            if ((age<18) && (age>=0)){
                    System.out.println(name+" "+surname+" usted es menor de edad, por lo tanto, no puede entrar a la fiesta, por favor devuélvase a su casa.");
                            }
                            if (age>=18){
                                System.out.println(name+" "+surname+" usted es mayor de edad, por lo tanto puede entrar a la fiesta.");
                            }
                        }
                    }
