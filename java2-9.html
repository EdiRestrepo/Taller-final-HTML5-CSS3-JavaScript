El profesor de geometría está explicando a sus estudiantes las fórmulas para calcular el área de diferentes figuras geométricas,
para ello requiere una aplicación que le facilite el ejercicio solicitándole los valores al estudiante. La aplicación debe permitir
que el estudiante seleccione si desea calcular el área de un rectángulo, triángulo o trapecio. No olvide solicitar los datos
necesarios para realizar cada cálculo y mostrar su respectivo resultado.

Ejercicio9

                package com.Ejercicio9;


                import java.io.IOException;
                import java.util.Scanner;
                
                public class Ejercicio9 {
                
                    private Scanner input = new Scanner(System.in);
                   private int option;
                   private boolean flag = true;
                
                   private double area,base,base2,height;
                
                    public void Ejercicio9 () throws IOException {
                
                        do {
                            menu();
                            option = captureDataInt();
                            input.nextLine(); //Limpiar el buffer de entrada después de leer el número entero
                            flag = evaluateOption(option);
                        }while (flag == true);
                    }
                
                    private void menu(){
                        System.out.println("App Geometría");
                        System.out.println("Por favor, digite un número de acuerdo a la figura");
                        System.out.println("1. Área de un rectángulo");
                        System.out.println("2. Área de un triángulo");
                        System.out.println("3. Área de un trapecio");
                        System.out.println("0. Salir de la App");
                    }
                
                    private void askBase(){
                        System.out.println("Por favor, digite la base en metros");
                    }
                    private void askBase2(){
                        System.out.println("Por favor, digite la base2 en metros");
                    }
                    private void askHeight(){
                        System.out.println("Por favor, digite la altura en metros");
                    }
                
                    private int captureDataInt(){
                        return input.nextInt();
                    }
                
                    private double captureDataDouble(){
                        return input.nextDouble();
                    }
                
                    private boolean evaluateOption(int option) throws IOException {
                        switch (option){
                            case 1:
                                rectangularArea();
                                System.out.println("Digita, enter para volver al menu principal");
                                System.in.read();
                                return true;
                            case 2:
                                triangleArea();
                                System.out.println("Digita, enter para volver al menu principal");
                                System.in.read();
                                return true;
                            case 3:
                                trapezoidArea();
                                System.out.println("Digita, enter para volver al menu principal");
                                System.in.read();
                                return true;
                            case 0:
                                return false;
                            //break;
                            default:
                                System.out.println("Opción incorrecta");
                                System.out.println("Digita, enter para volver al menu principal");
                                System.in.read();
                                return true;
                        }
                    }
                    private void captureData(){
                        askBase();
                        base = captureDataDouble();
                        askHeight();
                        height = captureDataDouble();
                    }
                
                /*Área rectángulo
                  Área de un rectangulo = base * altura
                  A = b*h (m^2)
                  donde A = área; b = base; h = altura
                 */
                    private void rectangularArea(){
                        captureData();
                        area=base*height;
                        System.out.println("El área del rectángulo es de "+area+"m^2");
                    }
                
                /*Área triángulo
                  Área de un triángulo = (base * altura)/2
                  A = (b*h)/2 (m^2)
                  donde A = área; b = base; h = altura
                 */
                    private void triangleArea(){
                        captureData();
                        area=(base*height)/2;
                        System.out.println("El área del triángulo es de "+area+"m^2");
                    }
                
                /*Área Trapecio
                  Área de un trapecio = ((base1+base2) * altura)/2.
                  A = ((b1+b2)*h)/2 (m^2)
                  donde A = área; b1 = base1; b2 = base2; h = altura
                 */
                    private void trapezoidArea(){
                        captureData();
                        askBase2();
                        base2 = captureDataDouble();
                        area=((base+base2)*height)/2;
                        System.out.println("El área del trapecio es de "+area+"m^2");
                    }
                }
