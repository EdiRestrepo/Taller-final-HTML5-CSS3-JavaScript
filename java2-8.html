Enunciado:
El pastelero Don Carlos es el mejor pastelero de la ciudad y requiere una aplicación que le permita registrar los pedidos de los clientes en cuanto a las tortas que realiza. Cada torta tiene unas características propias como sabor, cantidad (porciones) y decoraciones). Se requiere que la aplicación permita registrar los pedidos, las tortas disponibles y las ventas que se registren diariamente.

                package com.Ejercicio8;


                import java.io.IOException;
                import java.util.Scanner;
                
                public class Ejercicio8 {
                
                    private Scanner input = new Scanner(System.in);
                   private int option;
                   private boolean flag = true;
                   private String name;
                   private String[] flavor, decoration, portions;
                   private int[] priceCake, quantityCakes;
                
                   public void Ejercicio8 () throws IOException {
                       portions = new String[3];
                       flavor = new String[6];
                       decoration = new String[6];
                       priceCake = new int[3];
                       quantityCakes = new int[3];
                
                       priceCake[0]=10000;
                       priceCake[1]=18000;
                       priceCake[2]=25000;
                
                       portions[0]="Tortas pequeñas (4 porciones)";
                       portions[1]="Tortas medianas (8 porciones)";
                       portions[2]="Tortas grandes (12 porciones)";
                
                       flavor[0]="chocolate";
                       flavor[1]="vainilla";
                       flavor[2]="frutos rojos";
                
                       decoration[0]="arequipe";
                       decoration[1]="nutela";
                       decoration[2]="fresas";
                
                        do {
                            menu();
                            option = captureDataInt();
                            input.nextLine(); //Limpiar el buffer de entrada después de leer el número entero
                            flag = evaluateOption(option);
                        }while (flag == true);
                    }
                    private void menu(){
                        System.out.println("App tortas Don Carlos");
                        System.out.println("Por favor, digite un número de acuerdo a la acción que desea realizar");
                        System.out.println("1. Registrar pedido");
                        System.out.println("2. Ver tortas disponibles");
                        System.out.println("3. Registrar ventas del dia");
                        System.out.println("0. Salir de la App");
                    }
                
                    private void askName(){
                        System.out.println("Por favor, digite el nombre del cliente");
                    }
                    private void askCakeP(){
                        System.out.println("Por favor, digite la cantidad  de "+portions[0]);
                    }
                    private void askCakeM(){
                        System.out.println("Por favor, digite la cantidad  de "+portions[1]);
                    }
                    private void askCakeG(){
                        System.out.println("Por favor, digite la cantidad  de "+portions[2]);
                    }
                
                    private void askFlavor(){
                        System.out.println("Por favor, digite el sabor de la torta "+"("+flavor[0]+" - "+flavor[1]+" - "+flavor[2]+")");
                    }
                
                    private void askDecoration(){
                        System.out.println("Por favor, digite el sabor de la torta "+"("+decoration[0]+" - "+decoration[1]+" - "+decoration[2]+")");
                    }
                
                    private String captureDataString(){
                        return input.nextLine();
                    }
                    private int captureDataInt(){
                        return input.nextInt();
                    }
                
                    private boolean evaluateOption(int option) throws IOException {
                        switch (option){
                            case 1:
                                registerOrder();
                                System.out.println("Digita, enter para volver al menu principal");
                                System.in.read();
                                return true;
                            case 2:
                                availableCakes();
                                System.out.println("Digita, enter para volver al menu principal");
                                System.in.read();
                                return true;
                            case 3:
                                salesDay();
                                System.out.println("Digita, enter para volver al menu principal");
                                System.in.read();
                                return true;
                            case 0:
                                return false;
                            //break;
                            default:
                                System.out.println("Opción incorrecta");
                                System.out.println("Digita, enter para volver al menu principal");
                                System.in.read();
                                return true;
                        }
                    }
                
                    private void availableCakes() throws IOException {
                        System.out.println("TORTAS DISPINIBLES");
                        System.out.println("1. "+portions[0]+", $"+priceCake[0]);
                        System.out.println("2. "+portions[1]+", $"+priceCake[1]);
                        System.out.println("3. "+portions[2]+", $"+priceCake[2]);
                        System.out.println("SABORES: "+flavor[0]+" - "+flavor[1]+" - "+flavor[2]);
                        System.out.println("DECORACION: "+decoration[0]+" - "+decoration[1]+" - "+decoration[2]);
                    }
                
                    private void registerOrder(){
                        System.out.println("REGISTRAR UN PEDIDO");
                        askName();
                        name=captureDataString();
                        askCakeP();
                        quantityCakes[0] = captureDataInt();
                        input.nextLine(); //Limpiar el buffer de entrada después de leer el número entero
                        if(quantityCakes[0]>=1){
                            askFlavor();
                            flavor[3]=captureDataString();
                            askDecoration();
                            decoration[3]=captureDataString();
                        }
                        askCakeM();
                        quantityCakes[1] = captureDataInt();
                        input.nextLine(); //Limpiar el buffer de entrada después de leer el número entero
                        if(quantityCakes[1]>=1){
                            askFlavor();
                            flavor[4]=captureDataString();
                            askDecoration();
                            decoration[4]=captureDataString();
                        }
                        askCakeG();
                        quantityCakes[2] = captureDataInt();
                        input.nextLine(); //Limpiar el buffer de entrada después de leer el número entero
                        if(quantityCakes[2]>=1){
                            askFlavor();
                            flavor[5]=captureDataString();
                            askDecoration();
                            decoration[5]=captureDataString();
                        }
                        System.out.println("Su pedido se ha registrado de forma correcta");
                    }
                
                    private void salesDay(){
                        System.out.println("REGISTRAR LAS VENTAS DEL DIA");
                        askCakeP();
                        quantityCakes[0] = captureDataInt();
                        askCakeM();
                        quantityCakes[1] = captureDataInt();
                        askCakeG();
                        quantityCakes[2] = captureDataInt();
                        System.out.println("Total ventas del dia");
                        System.out.println("$ "+(quantityCakes[0]*priceCake[0]+quantityCakes[1]*priceCake[1]+quantityCakes[2]*priceCake[2]));
                    }
                
                }
