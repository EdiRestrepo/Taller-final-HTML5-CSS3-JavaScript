9. El profesor de geometría está explicando a sus estudiantes las fórmulas para calcular el área de diferentes figuras geométricas, para ello requiere una aplicación que le facilite el ejercicio solicitándole los valores al estudiante. La aplicación debe permitir que el estudiante seleccione si desea calcular el área de un rectángulo, triángulo o trapecio. No olvide solicitar los datos necesarios para realizar cada cálculo y mostrar su respectivo resultado. 

                    //Captura datos numericos

                    //Área Trapecio
                    // Área de un trapecio = ((base1+base2) * altura)/2.
                    // A = ((b1+b2)*h)/2 (m^2)
                    // donde A = área; b1 = base1; b2 = base2; h = altura
                    SubProceso areaTrapecio (a, b, c, d)
                        Limpiar Pantalla;
                        Escribir "Ingresa los siguientes datos para calcular el área de un trapecio";
                        Escribir "";
                        b<-capturaNumero("Ingresa la base1 del trapecio en metros");
                        d<-capturaNumero("Ingresa la base2 del trapecio en metros");
                        c<-capturaNumero("Ingresa la altura del trapecio en metros");
                        a<-((b+d)*c)/2;
                        
                        Limpiar Pantalla;
                        
                        Escribir "El área del trapecio es: ", a," m^2";
                        
                    FinSubProceso
                    
                    
                    //Área triángulo
                    // Área de un triángulo = (base * altura)/2
                    // A = (b*h)/2 (m^2)
                    // donde A = área; b = base; h = altura
                    SubProceso areaTriangulo (a, b, c Por Valor)
                        Limpiar Pantalla;
                        Escribir "Ingresa los siguientes datos para calcular el área de un triángulo";
                        Escribir "";
                        b<-capturaNumero("Ingresa la base del triángulo en metros");
                        c<-capturaNumero("Ingresa la altura del triángulo en metros");
                        a<-(b*c)/2;
                        Limpiar Pantalla;
                        
                        Escribir "El área del triángulo es: ", a," m^2";
                    FinSubProceso
                    
                    //Área rectángulo
                    // Área de un rectangulo = base * altura
                    // A = b*h (m^2)
                    // donde A = área; b = base; h = altura
                    SubProceso arearectangulo (a, b, c Por Valor)
                        Limpiar Pantalla;
                        Escribir "Ingresa los siguientes datos para calcular el área de un rectángulo";
                        Escribir "";
                        b<-capturaNumero("Ingresa la base del rectángulo en metros");
                        c<-capturaNumero("Ingresa la altura del rectángulo en metros");
                        a<-b*c;
                        Limpiar Pantalla;
                        
                        Escribir "El área del rectángulo es: ", a," m^2";
                        
                    FinSubProceso
