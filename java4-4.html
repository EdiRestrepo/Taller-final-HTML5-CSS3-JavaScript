                    package com.Ejercicio4;

                    import java.io.IOException;
                    import java.util.Scanner;
                    
                    public class Ejercicio4 {
                        private Scanner input = new Scanner(System.in);
                        private int option;
                        private boolean flag = true;
                    
                        private String [][] matrix;
                    
                        private int col;
                    
                        public void Ejercicio4 () throws IOException {
                            matrix = new String[4][5];
                            do {
                                menu();
                                option = captureDataInt();
                                input.nextLine(); //Limpiar el buffer de entrada después de leer el número entero
                                flag = evaluateOption(option);
                            }while (flag == true);
                        }
                    
                        private void menu(){
                            System.out.println("1. Matriz bidimensional");
                            System.out.println("0. Salir de la App");
                        }
                    
                        private int captureDataInt(){
                            return input.nextInt();
                        }
                    
                        private boolean evaluateOption(int option) throws IOException {
                            switch (option){
                                case 1:
                                    printArray();
                                    System.out.println("Digita, enter para volver al menu principal");
                                    System.in.read();
                                    return true;
                                case 0:
                                    return false;
                                //break;
                                default:
                                    System.out.println("Opción incorrecta");
                                    System.out.println("Digita, enter para volver al menu principal");
                                    System.in.read();
                                    return true;
                            }
                        }
                    
                        private void twoDimensionalArray(){
                            matrix[0][0]="01";
                            matrix[0][1]="02";
                            matrix[0][2]="03";
                            matrix[0][3]="04";
                            matrix[0][4]="05";
                    
                            matrix[1][0]="06";
                            matrix[1][1]="07";
                            matrix[1][2]="08";
                            matrix[1][3]="09";
                            matrix[1][4]="10";
                    
                            matrix[2][0]="11";
                            matrix[2][1]="12";
                            matrix[2][2]="13";
                            matrix[2][3]="14";
                            matrix[2][4]="15";
                    
                            matrix[3][0]="16";
                            matrix[3][1]="17";
                            matrix[3][2]="18";
                            matrix[3][3]="19";
                            matrix[3][4]="20";
                        }
                    
                        private void printArray(){
                            twoDimensionalArray();
                            for (int col = 0; col <= 4 ; col++) {
                                System.out.print(matrix[0][col]+" ");
                            }
                            System.out.println("");
                            for (int col = 4; col >= 0 ; col--) {
                                System.out.print(matrix[1][col]+" ");
                            }
                            System.out.println("");
                            for (int col = 0; col <= 4 ; col++) {
                                System.out.print(matrix[2][col]+" ");
                            }
                            System.out.println("");
                            for (int col = 4; col >= 0 ; col--) {
                                System.out.print(matrix[3][col]+" ");
                            }
                            System.out.println("");
                        }
                    }
