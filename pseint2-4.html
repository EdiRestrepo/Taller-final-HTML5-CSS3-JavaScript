Exercise 4

Enunciado:
La video tienda que presta sus servicios de alquiler de películas a los usuarios del barrio el Porvenir, requiere de una aplicación que permita registrar el alquiler de las películas que adquieren sus usuarios. Para cada usuario se debe permitir la opción de alquilar película, consultar películas disponibles y recibir película en la video tienda con la opción de realizar anotaciones sobre estas si se llegan a presentar daños u otra novedad sobre la película. 

                Proceso Taller2_4
                    Definir usuario, anotaciones Como Caracter;
                    Definir opcion, edad Como Entero;
                    
                    Escribir "APP video tienda alquiler de películas Porvenir";
                    Escribir  "";
                    Escribir "Ingrese su nombre de usuario";
                    Leer usuario;
                    Escribir "Ingrese su edad";
                    Leer edad;
                    
                    Limpiar Pantalla;
                    
                    Si edad >= 14 Entonces
                        Escribir "Hola ", usuario,", Bienvenido a su video tienda Porvenir";
                        Escribir "";
                        Escribir "Ingrese un número del 1 al 4 de acuerdo a lo que desea hacer";
                        Escribir "";
                        Escribir "1. Alquilar película";
                        Escribir "2. Consultar películas disponbles";
                        Escribir "3. Recibir película en la video tienda";
                        Escribir "4. Salir de la App";
                        Leer opcion;
                        
                        Limpiar Pantalla;
                        
                        Segun opcion Hacer
                            1:
                                Escribir "Lista de Películas disponibles";
                                Escribir "";
                                Escribir "1. Película 1";
                                Escribir "2. Película 2";
                                Escribir "3. Película 3";
                                Escribir "4. Película 4";
                                Escribir "5. Película 5";
                                Escribir "";
                                Escribir "Ingrese un número, de a acuerdo al nombre de la película que deseas alquilar";
                            
                                Leer opcion;
                                
                                Limpiar Pantalla;
                                
                                Segun opcion Hacer
                                    1: 
                                        Escribir "Hola ",usuario," ", "acabas de adquirir la Película 1, esperamos la disfrutes y regreses pronto"; 
                                    2: 
                                        Escribir "Hola ",usuario," ", "acabas de adquirir la Película 2, esperamos la disfrutes y regreses pronto";
                                    3: 
                                        Escribir "Hola ",usuario," ", "acabas de adquirir la Película 3, esperamos la disfrutes y regreses pronto"; 
                                    4: 
                                        Escribir "Hola ",usuario," ", "acabas de adquirir la Película 4, esperamos la disfrutes y regreses pronto"; 
                                    5: 
                                        Escribir "Hola ",usuario," ", "acabas de adquirir la Película 5, esperamos la disfrutes y regreses pronto"; 
                                    De Otro Modo:
                                        Escribir "La opción elegida no existe";
                                        Escribir "Intenta de nuevo";
                                FinSegun
                            2:
                                Escribir "Lista de Películas disponibles";
                                Escribir "";
                                Escribir "1. Película 1";
                                Escribir "2. Película 2";
                                Escribir "3. Película 3";
                                Escribir "4. Película 4";
                                Escribir "5. Película 5";
                                Escribir "";
                                Escribir "Regresa al menú principal si deseas alquilar alguna de nuestras películas";
                                
                            3:
                                Escribir "¿Alguna novedad al recibir la película (daño u otra novedad)?";
                                Escribir "";
                                Escribir "1. Si";
                                Escribir "2. No";
                                Escribir "";
                                Escribir "Digita 1 si hay alguna novedad, de lo contrario digita 2 para salir del sistema";
                                Leer opcion;
                                
                                Limpiar Pantalla;
                                
                                Si opcion = 1 Entonces
                                    Escribir "Digite cual es novedad";
                                    Leer anotaciones;
                                    Escribir "Se ha registrado la novedad: ", anotaciones;
                                    Escribir "Hasta pronto";
                                SiNo
                                    Escribir "Se ha guardado la información sin ninguna novedad";
                                    Escribir "Hasta pronto";
                                    
                                FinSi
                            4:
                                Escribir "El usuario ",usuario," Ha salido de la App";
                                
                            De Otro Modo:
                                Escribir "La opción elegida no existe";
                                Escribir "Intenta de nuevo";
                                
                        FinSegun
                        
                    SiNo
                        Escribir "Hola ",usuario," ", "lo sentimos, pero no esta disponible el alquiler de peliculas a menores de 14 años de edad";
                        
                    FinSi
                    
                    
                FinProceso
