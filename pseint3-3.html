Exercise 3

Enunciado:
Realizar la siguiente figura por medio del ciclo Repetir - Hasta Que:

                Proceso Taller3_3
                    Definir i, j, f, na, ne Como Entero;
                    Definir figura1, figura2 Como Caracter;
                    
                    figura1<-"*";
                    figura2<-" ";
                    
                    //número de asteriscos en cada fila
                    na<-1;
                    
                    // número de espacios en cada fila
                    ne<-20;
                    
                    //número de filas del arbol parte superior 
                    f<-11;
                    
                    // Inicio ciclo 1, con este primer ciclo se hace la parte superior (el triangulo) del arbol
                    i<-1;
                    Repetir
                        
                        // Inicio ciclo 2
                        j<-1;
                        Repetir
                            Escribir figura2 Sin Saltar;
                            j<-j+1;
                        Hasta Que j > ne;
                        
                        // Inicio ciclo 3
                        j<-1;
                        Repetir
                            Escribir figura1 Sin Saltar;
                            j<-j+1;
                        Hasta Que j>na;
                        
                        Escribir figura2;
                        i<-i+1;
                        na<-na+2;
                        ne<-ne-1;
                    Hasta Que i > f;
                
                    
                    // Tronco del arbol primera linea
                    
                    ne<-19;
                    na<-3;
                    i<-1;
                    f<-1;
                    
                    Repetir
                        
                      
                    j<-1;
                    Repetir
                        Escribir figura2 Sin Saltar;
                        j<-j+1;
                    Hasta Que j > ne;
                    
                    j<-1;
                    Repetir
                        Escribir figura1 Sin Saltar;
                        j<-j+1;
                    Hasta Que j>na;
                    
                    Escribir figura2;
                    i<-i+1;
                
                    Hasta Que i > f;
                    
                    // Tronco del arbol siguientes lineas
                    
                    ne<-19;
                    na<-3;
                    i<-1;
                    f<-3;
                    
                    Repetir
                        
                        
                        j<-1;
                        Repetir
                            Escribir figura2 Sin Saltar;
                            j<-j+1;
                        Hasta Que j > ne;
                        
                        j<-1;
                        Repetir
                            Escribir figura1 Sin Saltar;
                            j<-j+1;
                        Hasta Que j>na;
                        
                        Escribir figura2;
                        i<-i+1;
                        na<-na+2;
                        ne<-ne-1;
                    Hasta Que i > f;
                    Escribir "";
                    
                FinProceso
