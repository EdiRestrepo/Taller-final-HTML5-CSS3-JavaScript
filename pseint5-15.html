2. Crear un arreglo de números enteros de 20 posiciones, el cual, debe ser llenado con números aleatorios entre 1 y 100;
después de haber llenado dicho arreglo, se debe volver a recorrer utilizando un ciclo diferente al que se usó para
llenarse e imprimir los números pares e impares. Ejemplo

° Números pares: 2, 4, 6, 8, 10

° Números impares: 1, 3, 5, 7, 9

                    //Imprimir números impares
                    SubProceso imprimirImpares (a,b,c)
                        c<-0;
                        Escribir "Números impares: "; Sin Saltar;
                        Para a<-0 Hasta 19 Con Paso 1 Hacer
                            Si b[a] MOD 2 <> 0 Entonces
                                Mientras c=0 Hacer
                                    c<-c+1;
                                FinMientras
                                Si c = 1 Entonces
                                    Escribir b[a] Sin Saltar;
                                SiNo
                                    Escribir ", ",b[a] Sin Saltar;
                                FinSi
                                c<-c+1;
                            FinSi
                        FinPara
                        
                        Escribir "";
                        
                    FinSubProceso
                    
                    
                    //Imprimir números pares
                    SubProceso imprimirPares (a,b,c)
                        Escribir "Números pares: " Sin Saltar;
                        Para a<-0 Hasta 19 Con Paso 1 Hacer
                            Si b[a] MOD 2 = 0 Entonces
                                Mientras c=0 Hacer
                                    c<-c+1;
                                FinMientras
                                Si c = 1 Entonces
                                    Escribir b[a] Sin Saltar;
                                SiNo
                                    Escribir ", ",b[a] Sin Saltar;
                                FinSi
                                c<-c+1;
                            FinSi
                        FinPara
                        
                        Escribir "";
                    FinSubProceso
                    
                    
                    //Llenar Vector con números aleatorios del 1 al 100
                    SubProceso llenarVector (a,b)
                        Mientras a<20 Hacer
                            b[a]<-azar(99)+1;
                            a<-a+1;
                        FinMientras
                    FinSubProceso
