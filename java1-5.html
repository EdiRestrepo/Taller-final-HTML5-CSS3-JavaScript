Variables and Printing Java

Exercise 5

Enunciado:
Realizar un programa el cual solicite el nombre de su mascota, edad de la mascota, el tipo de mascota
y su nombre completo. Al finalizar el sistema debe e imprimir en pantalla el siguiente mensaje:
[Nombre de Mascota] es un(a) [Tipo de Mascota], el cual, tiene [Edad de la Mascota] años de edad y
[Nombre Completo] es actualmente su dueño(a).

Main

                    package com;

                    import com.Ejercicio1.Ejercicio1;
                    import com.Ejercicio2.Ejercicio2;
                    import com.Ejercicio3.Ejercicio3;
                    import com.Ejercicio4.Ejercicio4;
                    import com.Ejercicio5.Ejercicio5;
                    
                    import java.io.IOException;
                    import java.util.Scanner;
                    
                    public class Main {
                    
                        private static boolean flag = true;
                        private static int option;
                    
                        public static void main(String[] args) throws IOException {
                            do {
                                clearScreen();
                                menu();
                                option = captureOption();
                                flag = evaluateOption(option);
                            }while (flag == true);
                            System.exit(0);
                        }
                    
                        private static void menu() {
                            System.out.println("Escoja un ejercicio del taller");
                            System.out.println("1. Ejercicio #1");
                            System.out.println("2. Ejercicio #2");
                            System.out.println("3. Ejercicio #3");
                            System.out.println("4. Ejercicio #4");
                            System.out.println("5. Ejercicio #5");
                            System.out.println("0. Salir del taller");
                        }
                    
                        private static int captureOption(){
                            Scanner input = new Scanner(System.in);
                            return input.nextInt();
                        }
                    
                        private static boolean evaluateOption(int option) throws IOException {
                            switch (option){
                                case 1:
                                    Ejercicio1 ejercicio1 = new Ejercicio1();
                                    ejercicio1.Ejercicio1();
                                    return true;
                                    //break;
                                case 2:
                                    Ejercicio2 ejercicio2 = new Ejercicio2();
                                    ejercicio2.Ejercicio2();
                                    return true;
                                //break;
                                case 3:
                                    Ejercicio3 ejercicio3 = new Ejercicio3();
                                    ejercicio3.Ejercicio3();
                                    return true;
                                case 4:
                                    Ejercicio4 ejercicio4 = new Ejercicio4();
                                    ejercicio4.Ejercicio4();
                                    return true;
                                //break;
                                case 5:
                                    Ejercicio5 ejercicio5 = new Ejercicio5();
                                    ejercicio5.Ejercicio5();
                                    return true;
                                //break;
                                case 0:
                                    return false;
                                    //break;
                                default:
                                    System.out.println("Opción incorrecta");
                                    System.in.read();
                                    return true;
                            }
                    
                        }
                    
                        private static void clearScreen() {
                            System.out.print("\033[H\033[2J");
                            System.out.flush();
                        }
                    }

Ejercicio5

                    package com.Ejercicio5;

                    import java.util.Scanner;
                    
                    public class Ejercicio5 {
                    
                        private Scanner input;
                        private String petName;
                        private int petAge;
                        private String petType;
                    
                        private String fullName;
                    
                        public void Ejercicio5(){
                    
                            input = new Scanner(System.in);
                    
                            askPetName();
                            petName = captureDataString();
                    
                            askPetAge();
                            petAge = captureDataInt();
                    
                            askPetType();
                            petType = captureDataString();
                    
                            askFullName();
                            fullName = captureDataString();
                    
                            printMessage();
                        }
                    
                        private void askPetName(){
                            System.out.println("Por favor, digite el nombre de su mascota");
                        }
                        private void askPetAge(){
                            System.out.println("Por favor, digite la edad de su mascota");
                        }
                        private void askPetType(){
                            System.out.println("Por favor, digite el tipo mascota");
                        }
                        private void askFullName(){
                            System.out.println("Por favor, digite su nombre completo");
                        }
                    
                        private void printMessage(){
                    System.out.println(petName+" es un(a) "+petType+", el cual, tiene "+petAge+" años de edad y "+fullName+" es actualmente su dueño(a).");
                        }
                    
                        private String captureDataString(){
                            //Scanner input = new Scanner(System.in);
                            return input.nextLine().replace("\\n", "");
                        }
                    
                        private int captureDataInt(){
                            //Scanner input = new Scanner(System.in);
                            return Integer.parseInt(input.nextLine());}
                    
                    }
