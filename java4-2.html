Exercise 2

Enunciado:
Crear un arreglo de números enteros de 20 posiciones, el cual, debe ser llenado con números aleatorios entre 1 y 100;
después de haber llenado dicho arreglo, se debe volver a recorrer utilizando un ciclo diferente al que se usó para
llenarse e imprimir los números pares e impares. Ejemplo

° Números pares: 2, 4, 6, 8, 10

° Números impares: 1, 3, 5, 7, 9

                    package com.Ejercicio2;

                    import java.io.IOException;
                    import java.util.Random;
                    import java.util.Scanner;
                    
                    public class Ejercicio2 {
                        private Scanner input = new Scanner(System.in);
                        private int option;
                        private boolean flag = true;
                    
                        private int[] number;
                    
                        private Random random;
                    
                        private int i,j;
                    
                        public void Ejercicio2 () throws IOException {
                            number=new int[20];
                            random=new Random();
                            do {
                                menu();
                                option = captureDataInt();
                                input.nextLine(); //Limpiar el buffer de entrada después de leer el número entero
                                flag = evaluateOption(option);
                            }while (flag == true);
                        }
                    
                        private void menu(){
                            System.out.println("1. Crear un arreglo de números enteros de 20 posiciones, el cual,
                             debe ser llenado con números aleatorios entre 1 y 100");
                            System.out.println("0. Salir de la App");
                        }
                    
                        private int captureDataInt(){
                            return input.nextInt();
                        }
                    
                        private boolean evaluateOption(int option) throws IOException {
                            switch (option){
                                case 1:
                                    printNumber();
                                    System.out.println("");
                                    System.out.println("Digita, enter para volver al menu principal");
                                    System.in.read();
                                    return true;
                                case 0:
                                    return false;
                                //break;
                                default:
                                    System.out.println("Opción incorrecta");
                                    System.out.println("Digita, enter para volver al menu principal");
                                    System.in.read();
                                    return true;
                            }
                        }
                    
                        private void fillArray(){
                            i=0;
                            while (i<20){
                                number[i]=random.nextInt(100)+1;
                                i++;
                            }
                    /*        for (int i = 0; i < 20; i++) {
                                number[i]=random.nextInt(100)+1;
                            }*/
                        }
                    
                        private void printNumber(){
                            fillArray();
                            j=0;
                            System.out.print("Numeros pares: ");
                            for (int i = 0; i < 20; i++) {
                                if (number[i]%2==0){
                                    while (j==0){
                                        j++;
                                    }
                                    if(j==1){
                                        System.out.print(number[i]);
                                    }else {
                                        System.out.print(","+number[i]);
                                    }
                                    j++;
                                }
                            }
                            System.out.println("");
                            j=0;
                            System.out.print("Numeros impares: ");
                            for (int i = 0; i < 20; i++) {
                                if (number[i]%2!=0){
                                    while (j==0){
                                        j++;
                                    }
                                    if(j==1){
                                        System.out.print(number[i]);
                                    }else {
                                        System.out.print(","+number[i]);
                                    }
                                    j++;
                                }
                            }
                            System.out.println("");
                        }
                    }
