Exercise 6

Enunciado:
Se está creando una aplicación que va a estar conectada con un prototipo que permita almacenar contactos telefónicos en el dispositivo. Para ello cada contacto debe contener nombre completo, teléfono y organización. Se requiere que la aplicación permita añadir 3 contactos verificando que el número no esté almacenado, buscar contactos almacenados y eliminar contactos si el usuario lo requiere. Recuerde que el sistema debe terminar cuando el usuario así lo indique

                    package com.Ejercicio6;

                    import java.io.IOException;
                    import java.util.Scanner;
                    
                    public class Ejercicio6 {
                        private Scanner input = new Scanner(System.in);
                        private int option;
                        private boolean flag = true;
                        private String[] name,org;
                        private int[] phone;
                        private boolean[] memorySpace;
                    
                    
                        public void Ejercicio6 () throws IOException {
                            name = new String[3];
                            phone = new int[3];
                            org = new String[3];
                            memorySpace = new boolean[3];
                            memorySpace[0]=false;
                            memorySpace[1]=false;
                            memorySpace[2]=false;
                            name[0]="Contacto1";
                            name[1]="Contacto2";
                            name[2]="Contacto3";
                    
                            do {
                                menu();
                                option = captureDataInt();
                                input.nextLine(); //Limpiar el buffer de entrada después de leer el número entero
                                flag = evaluateOption(option);
                            }while (flag == true);
                        }
                    
                        private void menu(){
                            System.out.println("App para almacenar contactos telefónicos");
                            System.out.println("1. Añadir contacto");
                            System.out.println("2. Buscar contactos almacenados");
                            System.out.println("3. Eliminar contacto");
                            System.out.println("0. Salir de la App");
                        }
                    
                        private int captureDataInt(){
                            return input.nextInt();
                        }
                        private String captureDataString(){
                            return input.nextLine();
                        }
                    
                        private void askName(){
                            System.out.println("Digite el nombre completo del contacto");
                        }
                        private void askPhone(){
                            System.out.println("Digite el numero telefonico");
                        }
                        private void askOrg(){
                            System.out.println("Digite el nombre de la organización");
                        }
                    
                    
                        private boolean evaluateOption(int option) throws IOException {
                            switch (option){
                                case 1:
                                    addContact();
                                    System.out.println("Digita, enter para volver al menu principal");
                                    System.in.read();
                                    return true;
                                case 2:
                                    searchContact();
                                    System.out.println("Digita, enter para volver al menu principal");
                                    System.in.read();
                                    return true;
                                case 3:
                                    deleteContact();
                                    System.out.println("Digita, enter para volver al menu principal");
                                    System.in.read();
                                    return true;
                                case 0:
                                    return false;
                                //break;
                                default:
                                    System.out.println("Opción incorrecta");
                                    System.out.println("Digita, enter para volver al menu principal");
                                    System.in.read();
                                    return true;
                            }
                        }
                    
                        private void searchContact(){
                            if ((memorySpace[0]==false)&&(memorySpace[1]==false)&&(memorySpace[2]==false)){
                                System.out.println("Al momento no tiene contactos almacenados");
                            }else {
                                if(memorySpace[0]==true){
                                    System.out.println("1. Nombre del contacto: "+name[0]);
                                    System.out.println("   Numero telefonico: "+phone[0]);
                                    System.out.println("   Organizacion: "+org[0]);
                                }
                                if(memorySpace[1]==true){
                                    System.out.println("2. Nombre del contacto: "+name[1]);
                                    System.out.println("   Numero telefonico: "+phone[1]);
                                    System.out.println("   Organizacion: "+org[1]);
                                }
                                if(memorySpace[2]==true){
                                    System.out.println("3. Nombre del contacto: "+name[2]);
                                    System.out.println("   Numero telefonico: "+phone[2]);
                                    System.out.println("   Organizacion: "+org[2]);
                                }
                            }
                        }
                    
                        private void contactList1(){
                            if(memorySpace[0]==false){
                                System.out.println("1. "+name[0]);
                            }
                            if(memorySpace[1]==false){
                                System.out.println("2. "+name[1]);
                            }
                            if(memorySpace[2]==false){
                                System.out.println("3. "+name[2]);
                            }
                            System.out.println("4. Volver al menu principal");
                    
                        }
                        private void contactList2(){
                            if(memorySpace[0]==true){
                                System.out.println("1. "+name[0]);
                            }
                            if(memorySpace[1]==true){
                                System.out.println("2. "+name[1]);
                            }
                            if(memorySpace[2]==true){
                                System.out.println("3. "+name[2]);
                            }
                            System.out.println("4. Volver al menu principal");
                    
                        }
                    
                        private void deleteContact() throws IOException {
                            if ((memorySpace[0]==false)&&(memorySpace[1]==false)&&(memorySpace[2]==false)){
                                System.out.println("Al momento no tiene contactos almacenados");
                            }else {
                                System.out.println("Seleccione un numero de acuerdo al contacto que desea eliminar");
                                contactList2();
                                option = captureDataInt();
                                input.nextLine(); //Limpiar el buffer de entrada después de leer el número entero
                                switch (option){
                                    case 1:
                                        name[0]="Contacto1";
                                        phone[0]=0;
                                        org[0]="";
                                        memorySpace[0]=false;
                                        System.out.println("El Contacto1 se ha eliminado de forma correcta");
                                        break;
                                    case 2:
                                        name[1]="Contacto2";
                                        phone[1]=0;
                                        org[1]="";
                                        memorySpace[1]=false;
                                        System.out.println("El Contacto2 se ha eliminado de forma correcta");
                                        break;
                                    case 3:
                                        name[2]="Contacto3";
                                        phone[2]=0;
                                        org[2]="";
                                        memorySpace[2]=false;
                                        System.out.println("El Contacto3 se ha eliminado de forma correcta");
                                        break;
                                    case 4:
                                        break;
                                    default:
                                        System.out.println("Opción incorrecta");
                                        System.out.println("Digita, enter para volver al menu principal");
                                        System.in.read();
                                }
                            }
                        }
                    
                        private void addContact() throws IOException {
                            if ((memorySpace[0]==true)&&(memorySpace[1]==true)&&(memorySpace[2]==true)){
                                System.out.println("Espacio de memoria lleno, dedes de eliminar un contacto para agregar uno nuevo");
                            }else {
                                System.out.println("Seleccione un numero de acuerdo al contacto que desea añadir");
                                contactList1();
                                option = captureDataInt();
                                input.nextLine(); //Limpiar el buffer de entrada después de leer el número entero
                                switch (option){
                                    case 1:
                                        if(memorySpace[0] == false){
                                            askPhone();
                                            phone[0]=captureDataInt();
                                            input.nextLine(); //Limpiar el buffer de entrada después de leer el número entero
                                            if((phone[0]==phone[1])||(phone[0]==phone[2])){
                                                System.out.println("¡El numero telefonico que desea almacenar ya existe!");
                                                phone[0]=0;
                                            }else {
                                                askName();
                                                name[0]=captureDataString();
                                                askOrg();
                                                org[0]=captureDataString();
                                                memorySpace[0]=true;
                                            }
                                        }else {
                                            System.out.println("Ya tienes almacenado un contacto en esta posisición.");
                                            System.out.println("Puedes eliminar el contacto actual y almacenar un nuevo contacto");
                                        }
                                        break;
                                    case 2:
                                        if(memorySpace[1] == false){
                                            askPhone();
                                            phone[1]=captureDataInt();
                                            input.nextLine(); //Limpiar el buffer de entrada después de leer el número entero
                                            if((phone[0]==phone[1])||(phone[1]==phone[2])){
                                                System.out.println("¡El numero telefonico que desea almacenar ya existe!");
                                                phone[1]=0;
                                            }else {
                                                askName();
                                                name[1]=captureDataString();
                                                askOrg();
                                                org[1]=captureDataString();
                                                memorySpace[1]=true;
                                            }
                                        }else {
                                            System.out.println("Ya tienes almacenado un contacto en esta posisición.");
                                            System.out.println("Puedes eliminar el contacto actual y almacenar un nuevo contacto");
                                        }
                                        break;
                                    case 3:
                                        if(memorySpace[2] == false){
                                            askPhone();
                                            phone[2]=captureDataInt();
                                            input.nextLine(); //Limpiar el buffer de entrada después de leer el número entero
                                            if((phone[0]==phone[2])||(phone[1]==phone[2])){
                                                System.out.println("¡El numero telefonico que desea almacenar ya existe!");
                                                phone[2]=0;
                                            }else {
                                                askName();
                                                name[2]=captureDataString();
                                                askOrg();
                                                org[2]=captureDataString();
                                                memorySpace[2]=true;
                                            }
                                        }else {
                                            System.out.println("Ya tienes almacenado un contacto en esta posisición.");
                                            System.out.println("Puedes eliminar el contacto actual y almacenar un nuevo contacto");
                                        }
                                        break;
                                    case 4:
                                        break;
                                    default:
                                        System.out.println("Opción incorrecta");
                                        System.out.println("Digita, enter para volver al menu principal");
                                        System.in.read();
                                }
                            }
                        }
