Exercise 6

Enunciado:
El taller de motos "El Maquinista" recibe motocicletas de alto cilindraje para realizar las respectivas revisiones y requiere una aplicación que le permita registrar los servicios generados a sus clientes. Para cada motocicleta se debe tener registro del ingreso al taller y las observaciones por parte del cliente. También debe existir registro de salida del taller con las novedades y otra de arreglos hechos por el mecánico en caso de que se requiera inventariar cambios repuestos en la motocicleta al entregarla.

                    package com.Ejercicio6;


                    import java.io.IOException;
                    import java.util.Scanner;
                    
                    public class Ejercicio6 {
                    
                        private Scanner input = new Scanner(System.in);
                       private int option;
                       private boolean flag = true;
                       private String name;
                       private String customerComment, mechanicComment;
                       private String dateEntry, exitDate;
                       private String brandMotorcycle, plateMotorcycle;
                       private String typeMaintenance;
                    
                    
                        public void Ejercicio6 () throws IOException {
                            do {
                                menu();
                                option = captureDataInt();
                                flag = evaluateOption(option);
                            }while (flag == true);
                        }
                    
                        private void menu(){
                            System.out.println("App taller de motos *El Maquinista* ");
                            System.out.println("Por favor, digite un número de acuerdo a lo que desea hacer");
                            System.out.println("1. Registro ingreso motocicleta");
                            System.out.println("2. Registro salida motocicleta");
                            System.out.println("0. Salir de la App");
                        }
                    
                        private void askName(){
                            System.out.println("Por favor, digite su nombre de usuario");
                        }
                    
                        private String captureDataString(){
                            return input.nextLine().replace("\\n", "");
                        }
                    
                        private int captureDataInt(){
                            return Integer.parseInt(input.nextLine());
                        }
                    
                        private boolean evaluateOption(int option) throws IOException {
                            switch (option){
                                case 1:
                                    entryMotorcycle();
                                    System.out.println("Digita, enter para volver al menu principal");
                                    System.in.read();
                                    return true;
                                case 2:
                                    exitMotorcycle();
                                    System.out.println("Digita, enter para volver al menu principal");
                                    System.in.read();
                                    return true;
                                case 0:
                                    return false;
                                //break;
                                default:
                                    System.out.println("Opción incorrecta");
                                    System.out.println("Digita, enter para volver al menu principal");
                                    System.in.read();
                                    return true;
                            }
                        }
                    
                        private void entryMotorcycle(){
                            System.out.println("Digite los siguientes datos para registrar el ingreso de la motocicleta");
                            System.out.println("Fecha de ingreso de la motocicleta (dd-mm-aaaa)");
                            dateEntry = captureDataString();
                            System.out.println("Nombre completo del cliente");
                            name = captureDataString();
                            System.out.println("Marca de la moto (Yamaha, honda, kawasaki...)");
                            brandMotorcycle = captureDataString();
                            System.out.println("Placa de la moto");
                            plateMotorcycle = captureDataString();
                            System.out.println("Tipo de mantemiento a realizar (Preventivo, Correctivo)");
                            typeMaintenance = captureDataString();
                            System.out.println("Observaciones hechas por el clente al ingreso");
                            customerComment = captureDataString();
                            System.out.println("Se ha registrado la información de forma correcta");
                        }
                        private void exitMotorcycle(){
                            System.out.println("Digite los siguientes datos para registrar la salida de la motocicleta");
                            System.out.println("Fecha de salida de la motocicleta (dd-mm-aaaa)");
                            exitDate = captureDataString();
                            System.out.println("Nombre completo del cliente");
                            name = captureDataString();
                            System.out.println("Marca de la moto (Yamaha, honda, kawasaki...)");
                            brandMotorcycle = captureDataString();
                            System.out.println("Placa de la moto");
                            plateMotorcycle = captureDataString();
                            System.out.println("Tipo de mantemiento que se ha realizado (Preventivo, Correctivo)");
                            typeMaintenance = captureDataString();
                            System.out.println("Observaciones hechas por el mecanico (cambio de piezas, limpieza, reparación)");

                            System.out.println("Se ha registrado la información de forma correcta");
