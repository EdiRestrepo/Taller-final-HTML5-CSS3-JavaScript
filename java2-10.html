Enunciado:
El banco "Su banco fiel" es un banco que inicia sus actividades financieras y necesita una aplicación para llevar las cuentas de sus usuarios; por lo tanto, se sugiere que la cuenta tenga los atributos titular y cantidad. Para cada cliente las cuentas permitirán realizar ingresos, retiros o consultas de valor. En los ingresos no se pueden insertar valores negativos y para los retiros el valor no puede ser mayor al valor que tiene en la cuenta.

                package com.Ejercicio10;


                import java.io.IOException;
                import java.util.Scanner;
                
                public class Ejercicio10 {
                    private Scanner input = new Scanner(System.in);
                   private int option;
                   private boolean flag = true;
                   private String name;
                   private int amount, income, withdrawal;
                
                
                   public void Ejercicio10 () throws IOException {
                       amount = 0;
                       System.out.println("App *Su banco fiel*");
                       askName();
                       name = captureDataString();
                       do {
                            menu();
                            option = captureDataInt();
                            input.nextLine(); //Limpiar el buffer de entrada después de leer el número entero
                            flag = evaluateOption(option);
                        }while (flag == true);
                    }
                    private void menu(){
                        System.out.println(name+" por favor, digite un número de acuerdo a la acción que desea realizar");
                        System.out.println("1. Ingreso de dinero (deposito)");
                        System.out.println("2. Retiro de dinero");
                        System.out.println("3. Consulta de valor (saldo)");
                        System.out.println("0. Salir de la App");
                    }
                
                    private void askName(){
                        System.out.println("Por favor, digite el nombre del titular de la cuenta");
                    }
                
                    private String captureDataString(){
                        return input.nextLine();
                    }
                    private int captureDataInt(){
                        return input.nextInt();
                    }
                
                    private boolean evaluateOption(int option) throws IOException {
                        switch (option){
                            case 1:
                                cashInflow();
                                System.out.println("Digita, enter para volver al menu principal");
                                System.in.read();
                                return true;
                            case 2:
                                cashWithdrawal();
                                System.out.println("Digita, enter para volver al menu principal");
                                System.in.read();
                                return true;
                            case 3:
                                consultValue();
                                System.out.println("Digita, enter para volver al menu principal");
                                System.in.read();
                                return true;
                            case 0:
                                return false;
                            //break;
                            default:
                                System.out.println("Opción incorrecta");
                                System.out.println("Digita, enter para volver al menu principal");
                                System.in.read();
                                return true;
                        }
                    }
                    //En los ingresos no se pueden insertar valores negativos
                    private void cashInflow(){
                        System.out.println("Por favor, digita la cantidad de dinero que deseas deposiar");
                        income=captureDataInt();
                        if (income<0){
                            System.out.println("Lo sentimos pero la cantidad ingresada no es valida");
                        }else {
                            System.out.println("Su deposito fue de $"+income);
                            System.out.println("Su saldo actual es de $"+(amount=income+amount));
                        }
                    }
                    //para los retiros el valor no puede ser mayor al valor que tiene en la cuenta.
                    private void cashWithdrawal(){
                        System.out.println("Por favor, digita la cantidad de dinero que deseas retirar");
                        withdrawal=captureDataInt();
                        if (withdrawal>amount){
                            System.out.println("Lo sentimos pero la cantidad  que desea retirar supera el monto del valor de su cuenta");
                        }else {
                            System.out.println("Su retiro fue de $"+withdrawal);
                            System.out.println("Su saldo actual es de $"+(amount=amount-withdrawal));
                        }
                    }
                
                    private void consultValue(){
                        System.out.println("Su saldo actual en la cuenta es de $"+amount);
                    }
                }
